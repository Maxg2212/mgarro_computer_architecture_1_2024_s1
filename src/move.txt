#Funciones para pintar lineas
dibujo_horizontal:
     PINTO tx, ty
     SUM tx, tx, #1
     CE tx, fx
     BETO <>
     BRIN <>

dibujo_horizontal_izquierda:         
     PINTO tx,ty
     RES tx, tx, #1
     CE tx, fx
     BETO <>
     BRIN <> 

dibujo_vertical:
    PINTO tx,ty
    SUM ty, ty, #1
    CE ty, fy
    BETO <>
    BRIN <>

dibujo_vertical_arriba:
     PINTO tx,ty
     RES ty, ty, #1
     CE ty, fy
     BETO <>
     BRIN <>

dibujo_diagonal_abajo:
    PINTO tx, ty
    SUM tx, tx, #1
    SUM ty, ty, #1
    CE tx, fx
    BETO <sale>
    BRIN dibujo_diagonal_abajo

dibujo_diagonal_arriba:
    PINTO tx, ty
    SUM tx, tx, #1
    RES ty, ty, #1
    CE tx, fx
    BETO <sale>
    BRIN dibujo_diagonal_arriba
